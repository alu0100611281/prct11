def ==(other)
  if (@filas != other.filas || @columnas != other.columnas)
    return false
  end
  
  for i in 0...@filas
    
    for j in 0...@columnas
      if (@matriz[i][j] != other[i,j])
        return false
      end
    end
  end
  
  return true
end
